\paragraph{Sensitivity Definition:}
\begin{equation}
S_A^\alpha \equiv \frac{\partial \ln \nu_A}{\partial \ln \alpha} = \frac{\alpha}{\nu_A}\frac{\partial \nu_A}{\partial \alpha}.
\end{equation}

\paragraph{Optimal Pairs for \DFD Test:}
\begin{enumerate}
\item Cs -- Al$^+$: $\Delta S = 2.82$ (large baseline)
\item Yb$^+$ E3 -- Al$^+$: $\Delta S = -5.96$ (largest, opposite signs)
\item Cs -- Sr: $\Delta S = 2.77$ (readily available)
\end{enumerate}
