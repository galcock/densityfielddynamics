\section{DFD-CLASS Design Specification}

\textbf{Finding 236}: DFD-CLASS design document scoped.

\begin{tcolorbox}[colback=blue!5!white, colframe=dfdblue, title=\textbf{DFD-CLASS: Design Overview}]
\begin{itemize}[nosep]
\item \textbf{Base code}: CLASS v3.2.4 (Lesgourgues 2011; Blas, Lesgourgues, Tram 2011)
\item \textbf{Modifications}:
\begin{enumerate}[nosep]
\item New module: \texttt{dfd\_perturbations.c} --- implements DFD linear perturbation equations
\item Modified: \texttt{perturbations.c} --- adds $\psi$-field source terms to Einstein--Boltzmann system
\item Modified: \texttt{transfer.c} --- includes $\psi$-screen modification to luminosity distance
\item New module: \texttt{dfd\_nonlinear.c} --- halofit-calibrated DFD nonlinear power spectrum
\item Modified: \texttt{input.c} --- adds DFD parameter block (single parameter: enable/disable)
\end{enumerate}
\item \textbf{Validation}: Must reproduce all DFD predictions to $< 0.1\%$ relative to existing Python code
\item \textbf{Performance}: Must run in $< 2\times$ CLASS baseline time
\item \textbf{Timeline}: Alpha version by i98. Beta testing i99--i100.
\end{itemize}
\end{tcolorbox}

\section{Mock Catalogue Release Plan}

\textbf{Finding 237}: Mock catalogue public release specification.

\begin{itemize}[nosep]
\item \textbf{Contents}: 500k galaxies per mock, 100 realisations (DFD) + 100 realisations (GR), $z = 0$, 0.5, 1.0, 1.5
\item \textbf{Format}: HDF5 with standardised column names (position, velocity, mass, luminosity, type)
\item \textbf{HOD model}: Zheng et al.\ (2007) with DFD-calibrated parameters from N-body suite
\item \textbf{Covariance}: Jackknife covariance matrices included for each observable
\item \textbf{Documentation}: Complete README with column descriptions, HOD parameters, simulation specifications
\item \textbf{Repository}: Zenodo DOI + GitHub mirror
\item \textbf{Timeline}: Release concurrent with N-body paper I submission (i90)
\end{itemize}
