\section{Euclid Pipeline: Detailed Module Specifications}

\subsection{Module 1: Data Ingestion (\texttt{dfd-euclid-io})}
\begin{itemize}[nosep]
\item Input: Euclid DR1 public data products (FITS/HDF5 format)
\item Output: Standardised DFD-format catalogues
\item Dependencies: \texttt{astropy}, \texttt{h5py}, \texttt{healpy}
\item Validation: Reproduce Euclid team's published $N(z)$ and footprint area
\end{itemize}

\subsection{Module 3: $E_G$ Analysis (\texttt{dfd-euclid-eg})}
\begin{itemize}[nosep]
\item Input: Spectroscopic galaxy catalogue + photometric shear catalogue
\item Algorithm: Compute $E_G(k,z)$ in 6 $z$-bins using the Reyes et al.\ (2010) estimator
\item DFD-specific: Fit $\epsilon_\psi(k)$ profile in each $z$-bin; test $k^2$ scaling
\item Output: $\epsilon_\psi$ constraints per bin + combined significance
\item Validation: Apply to DFD mock catalogues; recover input $\psi_0$ to within $1\sigma$
\end{itemize}

\subsection{Module 6: Likelihood (\texttt{dfd-euclid-like})}
\begin{itemize}[nosep]
\item Input: All Module 2--5 outputs + covariance matrices
\item Algorithm: Bayesian inference on $\psi_0$ (and nuisance parameters) using MultiNest/emcee
\item DFD model: Single additional parameter $\psi_0/M_{\rm Pl}$ over $\Lambda$CDM
\item Priors: Flat on $\psi_0/M_{\rm Pl} \in [0, 10^{-4}]$; Gaussian on nuisance parameters
\item Output: Posterior on $\psi_0$, Bayes factor DFD/$\Lambda$CDM, model comparison
\item Validation: Inject DFD signal into $\Lambda$CDM mocks; verify recovery
\end{itemize}
