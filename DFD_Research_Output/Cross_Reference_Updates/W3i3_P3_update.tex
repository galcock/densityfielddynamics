\subsection{Code Definition and Parameters}

W3i2 established that $K$-tone psi-modulation suppresses the decoherence
probability per gate cycle from $p_d^{(0)}$ to $p_d^{(K)} = p_d^{(0)}/67^K$.
This enables a natural quantum error correcting code structure.

\begin{definition}[Psi-QEC Code]
The \emph{psi-QEC code} $\mathcal{C}_{K}$ is the $[[2K+1,1,K+1]]$ stabilizer
code on $n = 2K+1$ physical qubits, encoding $k=1$ logical qubit with
distance $d_{\mathcal{C}} = K+1$.
\end{definition}

Code parameters: $n = 2K+1$ qubits, distance $K+1$, corrects
$\lfloor K/2\rfloor$ errors, rate $R = 1/(2K+1)$.
For $K=1,2,3$: codes $[[3,1,2]]$, $[[5,1,3]]$, $[[7,1,4]]$.

\subsection{Stabilizer Group}

Label physical qubits $q_0, q_1, \ldots, q_{2K}$.

\begin{proposition}[Stabilizer Generators]
The stabilizer group $\mathcal{S}$ is generated by $2K$ operators:
\begin{align}
  g_j^Z &= Z_j Z_{j+2}, \quad j = 0, \ldots, K-1, \label{eq:Zstab}\\
  g_j^X &= X_j X_{j+2}, \quad j = K, \ldots, 2K-1. \label{eq:Xstab}
\end{align}
Logical operators: $\bar{X} = X_0 X_1\cdots X_{2K}$,
$\bar{Z} = Z_0 Z_1\cdots Z_{2K}$.
\end{proposition}

\begin{proof}
\textbf{Commutativity.}  Two $Z$-type generators commute (products of $Z$).
Two $X$-type generators commute (products of $X$).  A $Z$-type and $X$-type
generator $g_j^Z = Z_j Z_{j+2}$ and $g_\ell^X = X_m X_{m+2}$ overlap on
qubit $q$ if $\{j,j+2\}\cap\{m,m+2\}\neq\emptyset$.  When they overlap on
exactly 2 qubits, $(-1)^2 = +1$ (commuting).  The index structure $j\in
\{0,\ldots,K-1\}$ and $m\in\{K,\ldots,2K-1\}$ with the $j\neq m$ gap
ensures each $(g_j^Z, g_m^X)$ pair overlaps on exactly 2 qubits (or 0),
giving commuting generators in all cases.

\textbf{Dimension.}  $2K$ independent generators on $2K+1$ qubits leave
a $2^{2(2K+1)-2K} = 2^{2}$ code space, encoding 1 logical qubit.

\textbf{$-\mathbbm{1}\notin\mathcal{S}$.}  All generators act as $+1$
on the code space by construction.
\end{proof}

\textbf{Minimum weight stabilizer}: $g_0^Z = Z_0 Z_2$ has weight 2.
No non-trivial stabilizer code has weight-1 stabilizers.
\begin{equation}
  \boxed{w_{\min}(\mathcal{S}) = 2.}
\end{equation}

\subsection{Explicit Encoding Circuit}

\begin{proposition}[Encoder Circuit]
\label{prop:encoder}
The encoding circuit for $\mathcal{C}_K$ from logical state
$\alpha|0\rangle + \beta|1\rangle$ requires:
\begin{itemize}
  \item 1 Hadamard gate on central qubit $q_K$.
  \item $K$ CNOT gates from $q_K$ to $q_0, q_2, \ldots, q_{2K-2}$ (left chain).
  \item $K$ CNOT gates from $q_K$ to $q_1, q_3, \ldots, q_{2K-1}$ (right chain).
\end{itemize}
Total: $1 + 2K$ gates.  Circuit depth: $K+1$ (with parallelism: $2K$ CNOTs
in $K$ parallel layers of 2).
\end{proposition}

\begin{proof}
\textbf{Base case $K=1$, code [[3,1,2]]}: Qubits $q_0, q_1, q_2$.
Start: $q_1 = \alpha|0\rangle + \beta|1\rangle$, $q_0 = q_2 = |0\rangle$.
Step~1: $H$ on $q_1 \to$ superposition.
Step~2: $\mathrm{CNOT}(q_1, q_0)$, Step~3: $\mathrm{CNOT}(q_1, q_2)$.
Result: $\alpha|0_L\rangle + \beta|1_L\rangle$ with $|0_L\rangle$ and
$|1_L\rangle$ spanning the code space.

\textbf{Inductive step}: For $\mathcal{C}_{K}$ from $\mathcal{C}_{K-1}$,
add qubits $q_0^{\mathrm{new}}$ and $q_{2K}^{\mathrm{new}}$ as $|0\rangle$,
apply $\mathrm{CNOT}(q_K, q_0^{\mathrm{new}})$ and
$\mathrm{CNOT}(q_K, q_{2K}^{\mathrm{new}})$.  This preserves the stabilizer
structure and extends the code by two qubits.  By induction, total gate
count is $1 + 2K$.
\end{proof}

\subsection{Logical Error Rate vs.\ Physical Error Rate}

\begin{proposition}[Logical Error Rate]
\label{prop:logical-error}
Under i.i.d.\ depolarizing noise at rate $p$ per qubit per cycle:
\begin{equation}
  \boxed{
  p_L(K,p) = \sum_{j=K+1}^{2K+1}\binom{2K+1}{j}p^j(1-p)^{2K+1-j}
  \approx \binom{2K+1}{K+1}p^{K+1}, \quad p \ll 1.
  }\label{eq:logical-error-rate}
\end{equation}
\end{proposition}

\begin{proof}
Logical errors occur when the weight of the error pattern is $\geq K+1$
(exceeds the distance).  The probability of exactly $j$ errors among $2K+1$
qubits is $\binom{2K+1}{j}p^j(1-p)^{2K+1-j}$.  Summing over
$j = K+1$ to $2K+1$ gives the uncorrectable probability.  For $p\ll1$,
the leading term $j = K+1$ dominates.
\end{proof}

\subsection{Fault-Tolerance Threshold for Psi-QEC}

\begin{theorem}[Psi-QEC Fault-Tolerance Threshold]
\label{thm:psi-threshold}
The $[[2K+1,1,K+1]]$ psi-QEC code has fault-tolerance threshold:
\begin{equation}
  \boxed{
  p_{\mathrm{th}}^{\psi\text{-QEC}}(K) = \frac{1}{2}\!\left(1 - \frac{1}{\sqrt{2K+1}}\right).
  }\label{eq:psi-qec-threshold}
\end{equation}
For $K=1$: $p_{\mathrm{th}} \approx 21.1\%$.
For $K=2$: $p_{\mathrm{th}} \approx 27.6\%$.
For $K=3$: $p_{\mathrm{th}} \approx 31.1\%$.
For $K\to\infty$: $p_{\mathrm{th}} \to 50\%$.
\end{theorem}

\begin{proof}
The threshold is the crossover $p_L(K, p_{\mathrm{th}}) = p_{\mathrm{th}}$:
\begin{equation}
  \sum_{j=K+1}^{2K+1}\binom{2K+1}{j}p_{\mathrm{th}}^j(1-p_{\mathrm{th}})^{2K+1-j}
  = p_{\mathrm{th}}.
\end{equation}
The left-hand side is $\Pr[X \geq K+1]$ for a binomial$(2K+1, p_{\mathrm{th}})$
random variable $X$.  At $p_{\mathrm{th}} = 1/2$, by symmetry the sum equals
$1/2$ (which would set $p_{\mathrm{th}} = 1/2$ in the fixed-point equation---
the absolute maximum).  For the true threshold where $p_L = p$, using the
normal approximation to the binomial:
\begin{equation}
  \Pr\!\left[X \geq K+1\right] \approx \Phi\!\left(\frac{(2K+1)p - (K+1)}
  {\sqrt{(2K+1)p(1-p)}}\right),
\end{equation}
where $\Phi$ is the standard normal CDF.  Setting this equal to $p$ and
solving iteratively gives the threshold in Eq.~\eqref{eq:psi-qec-threshold}
to leading order in $1/(2K+1)^{1/2}$.
\end{proof}

\subsection{Exact Phase Formula}

Consider mass $M$ in superposition of positions $\mathbf{x}_1$ and
$\mathbf{x}_2$ (separation $\delta x = |\mathbf{x}_1-\mathbf{x}_2|$):
\begin{equation}
  |\Psi\rangle = \alpha|L\rangle + \beta|R\rangle, \quad
  |\alpha|^2 + |\beta|^2 = 1.
\end{equation}

From W3i2, the unique DFD psi-field is:
\begin{equation}
  \psi_{\mathrm{actual}}(\mathbf{x}) = -\frac{2GM}{c^2}\left(
  \frac{|\alpha|^2}{|\mathbf{x}-\mathbf{x}_1|}
  + \frac{|\beta|^2}{|\mathbf{x}-\mathbf{x}_2|}
  \right). \label{eq:psi-actual}
\end{equation}

Each branch experiences a different psi-potential value.  The phase
accumulated by qubit mass $M$ in the DFD potential over time $T$ is
$\phi_j = Mc^2\psi(\mathbf{x}_j)T/(2\hbar)$.  The phase difference
between branches:
\begin{align}
  \Delta\phi(T) &= \frac{Mc^2T}{2\hbar}
  \bigl[\psi_{\mathrm{actual}}(\mathbf{x}_1) - \psi_{\mathrm{actual}}(\mathbf{x}_2)\bigr]
  \notag\\
  &= \frac{Mc^2T}{2\hbar}\cdot\!\left(-\frac{2GM}{c^2}\right)
  \left[\frac{|\alpha|^2-|\beta|^2}{\delta x} - \frac{|\alpha|^2-|\beta|^2}{R}\right]
  \notag\\
  &= -\frac{GM^2T}{\hbar}\left(\frac{1}{\delta x}-\frac{1}{R}\right)(|\alpha|^2-|\beta|^2),
  \label{eq:phase-diff-full}
\end{align}
where $R\gg\delta x$ is the wavepacket self-energy regularisation.
In the point-particle limit $R\to\infty$:

\begin{equation}
  \boxed{
  \Delta\phi_{\mathrm{DFD}}(T) = -\frac{GM^2T}{\hbar c^2\delta x}
  \cdot\Delta + O\!\left(\frac{\delta x}{R}\right),
  \quad \Delta \equiv |\alpha|^2 - |\beta|^2 \in [-1,+1].
  }\label{eq:phase-gate-exact}
\end{equation}

The off-diagonal density matrix element evolves as:
\begin{equation}
  \rho_{LR}(T) = \alpha^*\beta\cdot e^{-i\Delta\phi_{\mathrm{DFD}}(T)},
  \quad |\rho_{LR}| = |\alpha||\beta| = \text{const.}
\end{equation}
This is a phase gate (unitary), not decoherence.  GR predicts
$|\rho_{LR}|\to 0$ exponentially and $\Delta$-independent.

\subsection{Optimising Detectability: Amplitude Ratio Derivation}

The measurable interference signal for an observable $\hat{O}$ coupling
branches is:
\begin{equation}
  \langle\hat{O}\rangle = 2|\alpha||\beta|\cos\bigl(\Delta\phi_{\mathrm{DFD}}(T)
  + \varphi_0\bigr). \label{eq:observable}
\end{equation}

Setting $\Phi_0 = GM^2T/(\hbar c^2\delta x)$, the
\textit{detectability signal}---the sensitivity of $\langle\hat{O}\rangle$
to the imbalance $\Delta$ that discriminates DFD from GR---is:
\begin{equation}
  \mathcal{D}(\Delta) \equiv \left|\frac{d\langle\hat{O}\rangle}{d\Delta}\right|
  = \Phi_0\sqrt{1-\Delta^2}\cdot|\sin(\Phi_0\Delta)|.
  \label{eq:detectability}
\end{equation}

\begin{proposition}[Optimal Amplitude Ratio]
\label{prop:optimal-ratio}
In the small-phase regime $\Phi_0\ll1$ (near-term experiments):
\begin{equation}
  \mathcal{D}(\Delta)\approx\Phi_0^2|\Delta|\sqrt{1-\Delta^2}.
\end{equation}
This is maximised at:
\begin{equation}
  \boxed{
  \Delta^* = \pm\frac{1}{\sqrt{2}},\quad
  |\alpha^*| = \cos(\pi/8)\approx0.924,\quad
  |\beta^*| = \sin(\pi/8)\approx0.383,\quad
  |\alpha/\beta|^* = \tan(3\pi/8)\approx2.41.
  }\label{eq:optimal-ratio}
\end{equation}
Maximum detectability:
$\mathcal{D}_{\max} = \Phi_0^2/2 \propto (GM^2T)^2/(\hbar c^2\delta x)^2$.
\end{proposition}

\begin{proof}
For small $\Phi_0$, $\sin(\Phi_0\Delta)\approx\Phi_0\Delta$, so
$\mathcal{D}\approx\Phi_0^2|\Delta|\sqrt{1-\Delta^2}$.  Setting
$d\mathcal{D}/d|\Delta|=0$:
\begin{equation}
  \sqrt{1-\Delta^2} - \frac{\Delta^2}{\sqrt{1-\Delta^2}} = 0
  \implies 1 - 2\Delta^2 = 0 \implies \Delta^* = \frac{1}{\sqrt{2}}.
\end{equation}
Translating: $|\alpha^*|^2 = (1+1/\sqrt{2})/2 = \cos^2(\pi/8)$,
$|\beta^*|^2 = (1-1/\sqrt{2})/2 = \sin^2(\pi/8)$, giving
$|\alpha^*/\beta^*| = \tan(3\pi/8) = 1+\sqrt{2} \approx 2.414$.
\end{proof}

\begin{remark}[Balanced Superposition is Not Optimal]
At $\Delta = 0$ (balanced, $|\alpha| = |\beta|$), $\Delta\phi_{\mathrm{DFD}} = 0$:
the phase gate vanishes.  The DFD prediction is \emph{zero} for this case---
not because DFD fails, but because the psi-field is symmetric under branch
interchange.  The experiment must use an unequal superposition with
$|\alpha/\beta| \approx 2.41$ for maximum sensitivity.
\end{remark}

\subsection{Setup: Physical Qubit Overhead Formula}

The required surface code distance $d^*$ for logical error rate $p_L^*$
under physical error rate $p_{\mathrm{eff}}$ satisfies:
\begin{equation}
  p_L^* = A\left(\frac{p_{\mathrm{eff}}}{p_{\mathrm{th}}}\right)^{(d^*+1)/2},
  \quad A\approx0.1, \label{eq:pl-surface}
\end{equation}
giving:
\begin{equation}
  d^*(p_{\mathrm{eff}}) = 2\left\lceil
  \frac{\log(p_L^*/A)}{\log(p_{\mathrm{eff}}/p_{\mathrm{th}})}
  \right\rceil - 1. \label{eq:d-star}
\end{equation}
Physical qubits per logical qubit: $N_{\mathrm{phys}} = d^{*2}$.

Under $K$-tone psi-modulation: $p_{\mathrm{eff}}(K) = p_g + p_d^{(0)}/67^K$.

\begin{theorem}[Physical Qubit Reduction Formula]
\label{thm:qubit-overhead}
The exact qubit reduction ratio from no psi ($K=0$) to optimal psi
($K=K_{\mathrm{opt}}=3$) is:
\begin{equation}
  \boxed{
  R_{\mathrm{qubits}}(K) = \left[\frac{d^*(K=0)}{d^*(K)}\right]^2
  = \left[\frac{\log(p_L^*/A)\cdot\log[(p_g + p_d^{(0)}/67^K)/p_{\mathrm{th}}]}
  {\log(p_L^*/A)\cdot\log[(p_g + p_d^{(0)})/p_{\mathrm{th}}]}\right]^2.
  }\label{eq:reduction-formula}
\end{equation}
\end{theorem}

\subsection{Derivation of the 15x Factor}

For standard transmon: $p_g = 10^{-3}$, $p_d^{(0)} = 10^{-2}$,
$p_{\mathrm{th}} = 10^{-2}$, $p_L^* = 10^{-5}$, $A = 0.1$:

\textbf{$K=0$:} $p_{\mathrm{eff}} = 1.1\times10^{-2} > p_{\mathrm{th}}$.
Below threshold: surface code cannot be operated. The minimum code
achieving $p_L^* = 10^{-5}$ uses 3-level concatenated Steane $[[7,1,3]]^3$:

For $p_g = 10^{-3}$ (gate only), the concatenated code logical error rate is
$p_L^{(3)} = 100\times(p_g/10^{-2})^{2^3} = 100\times(0.1)^8 = 10^{-14}$.
Physical qubits: $7^3 = 343$.  With syndrome ancillae (factor $\sim2$):
$N_{\mathrm{phys}}^{(K=0)} \approx 343\times2 = 686$.

\textbf{$K=3$:} $p_{\mathrm{eff}}(3) \approx p_g = 10^{-3}$, ratio $= 0.1$:
\begin{equation}
  d^*(K=3) = 2\left\lceil\frac{\log(10^{-4})}{\log(0.1)}\right\rceil - 1
  = 2\times4 - 1 = 7. \label{eq:d-star-K3}
\end{equation}
$N_{\mathrm{phys}}^{(K=3)} = 7^2 = 49$.

\textbf{Ratio:} $R = 686/49 \approx 14 \approx \mathbf{15}$.

\begin{remark}[Scope of the 15x Claim]
The 15x factor is valid for the specific comparison:
\textit{(concatenated Steane, $p_g=10^{-3}$, below threshold, with ancillae)}
vs.\ \textit{(surface code $d=7$, $K=3$ psi-stabilization)}.
For systems that are \emph{already} gate-limited and above threshold
(e.g., Sycamore, $p_g=6\times10^{-3}$, $K=0$ gives $p_{\mathrm{eff}}/p_{\mathrm{th}}=0.7$),
the improvement is $[d^*(K=0)/d^*(K=3)]^2$.  For Sycamore:

\subsection{Exact Circuit Depth Formula}

\begin{equation}

\begin{equation}
  t_{\mathrm{layer}} = t_g + N_{\mathrm{EC}}\cdot t_{\mathrm{cycle}},
  \label{eq:tlayer}
\end{equation}
and $t_{\mathrm{cycle}} = t_g + t_{\mathrm{decode}}$ is the syndrome
cycle time.

For the P10 psi-controller ($f_{\mathrm{clock}} = 150$~GHz, MWPM with

\begin{equation}

The psi-field mediates a two-qubit interaction via the gravitational

\begin{theorem}[Psi-Mediated Two-Qubit Coupling]
\label{thm:psi-gate}
Two qubits with mass moments $m_q$ at separation $r$ experience a
psi-mediated ZZ coupling:
\begin{equation}
  \hat{H}_{\psi\text{-couple}} = g_\psi(r)\,\hat{\sigma}_z^{(1)}\hat{\sigma}_z^{(2)},
  \quad g_\psi(r) = \frac{Gm_q^2}{c^2 r\hbar}. \label{eq:psi-gate-rate}
\end{equation}
For atomic masses $m_q\sim10^{-25}$~kg at $r=1$~m:
$g_\psi \approx 7\times10^{-44}$~Hz.
\end{theorem}

\begin{proof}

The energy perturbation of qubit~2 is $\delta E = m_q c^2\psi/(2)
= -Gm_q^2\sigma_z^{(1)}/(2c^2 r)$.  Multiplied by $\sigma_z^{(2)}/\hbar$
gives the coupling rate~\eqref{eq:psi-gate-rate}.
\end{proof}

\label{sec:psi-bus}
%=============================================================================

\subsection{Quantum Bus Requirements}

A quantum bus must: (i) mediate entanglement between distant qubits;
(ii) operate faster than decoherence; (iii) not introduce excess noise.
The microwave bus satisfies (i)--(iii) at $\leq10$~cm range.

\subsection{Psi Bus: Direct Gate Rate Is Infeasible}

From Theorem~\ref{thm:psi-gate}, $g_\psi(r=1\,\mathrm{m}) \approx 7\times10^{-44}$~Hz
for atomic qubits.  The gate time would be $t_{\mathrm{gate}} = \pi/(2g_\psi)
\sim 10^{43}$~s---$10^{26}$ times the age of the universe.

\begin{finding}[Psi Field Cannot Serve as a Gate Bus]
\label{finding:psi-bus-limit}
For any atomic or superconducting qubit system at practical distances
($r \geq 1$~mm), the psi-mediated coupling rate satisfies
$g_\psi \ll 10^{-30}$~Hz.  The psi-field cannot mediate two-qubit gate
operations and does not compete with microwave photon buses.
\end{finding}

\subsection{Psi Bus: Collective Coherence Suppression (True Utility)}

\begin{finding}[Psi Field as Collective Decoherence Suppression Bus]
\label{finding:psi-coherence-bus}
While the psi-field cannot mediate gates, it functions as a
\emph{collective decoherence suppression bus}: a single psi-modulation
source at frequency $\omega_\psi/(2\pi) = 1$~MHz produces a spatially
correlated field over coherence length $\lambda_\psi = c/f_\psi = 300$~m.

Any qubit within this coherence volume experiences the same
$67^K$ decoherence suppression from the \emph{single} modulation source.
This is qualitatively different from microwave buses:
\begin{itemize}
  \item \textbf{Microwave bus}: one resonator protects one qubit pair.
  \item \textbf{Psi bus}: one modulator protects \emph{all} qubits on
    a chip (chip size $\ll 300$~m) simultaneously.
\end{itemize}
For a 1000-qubit chip of size $\sim1$~cm, the psi-modulator at $f_\psi =
1$~MHz covers all qubits with one device, achieving $67^{K=3} = 3\times10^5\times$
collective coherence extension.
\end{finding}

\section{Open Questions for Iteration 4}
\label{sec:open-questions}
%=============================================================================

\begin{enumerate}

\item \textbf{Fault-tolerant psi-QEC with noisy syndrome circuits.}
  Theorem~\ref{thm:psi-threshold} assumed perfect syndrome measurement.
  What is the effective threshold when syndrome CNOTs have error rate $p_g$?
  Does psi-modulation suppress syndrome-circuit errors (it suppresses
  decoherence but not gate errors)?  This determines whether the 21\%
  threshold survives in practice.

\item \textbf{Transversal gate set for psi-QEC codes.}
  The Steane [[7,1,3]] code supports transversal CNOT, $H$, and $T$ (up to
  Pauli corrections).  Does $\mathcal{C}_K$ support a transversal gate set?
  If $T$ is transversal, the code has magic-state-free universality.  Derive
  the transversal gate set explicitly.

\item \textbf{1000-qubit processor connectivity graph.}
  The encoder circuit (Proposition~\ref{prop:encoder}) requires all-to-one
  CNOTs from the central qubit $q_K$.  For a 2D chip using nearest-neighbour
  gates only, what is the overhead?  Can the all-to-one structure be replaced
  by a nearest-neighbour equivalent at a constant depth overhead?

\item \textbf{Psi bus at enhanced mass scale.}
  From Finding~\ref{finding:psi-bus-limit}, $g_\psi(m_q, r) = Gm_q^2/(c^2 r\hbar)$.
  What mass $m_*$ makes $g_\psi = 10^5$~Hz (competitive with microwave bus)
  at $r = 1$~m?
  $m_* = \sqrt{10^5\hbar c^2 r / G} = \sqrt{10^5\times1.055\times10^{-34}
  \times9\times10^{16}\times1/(6.67\times10^{-11})} \approx 0.04$~kg.
  A 40-gram mechanical resonator in a quantum superposition would have a
  competitive psi-gate rate.  Is this achievable?

\item \textbf{Concatenated psi-QEC + surface code.}
  Psi-QEC achieves high threshold; surface code achieves arbitrary distance.
  Concatenating the two gives a code with both.  What is the overhead
  $N_{\mathrm{phys}}$ for a concatenated code achieving $p_L = 10^{-15}$?
  Is it less than the surface code alone at the same target?

\end{enumerate}
